script de création des icones et des alias dans le fichier .bashrc sur

#!/bin/bash


echo ""  >> ~/.bashrc
echo "alias looping='wine /opt/looping-mcd/Looping.exe'" >> ~/.bashrc
echo "alias datagrip='/opt/datagrip/bin/datagrip.sh'" >> ~/.bashrc
echo "alias pycharm='/opt/pycharm/bin/pycharm.sh'" >> ~/.bashrc 



fichier="datagrip.desktop"
nom_appli="datagrip"
chemin="/opt/DataGrip/bin"
nom_exec="datagrip.sh"
dest="/home/$USER/Bureau"
#dest="~/.config/mate/panel2.d/default/launchers"
echo "${dest}/${fichier}"
touch ${dest}/${fichier}
chmod 755 "${dest}/${fichier}"

echo "#!/usr/bin/env xdg-open" > "${dest}/${fichier}"
echo "[Desktop Entry]" >> "${dest}/${fichier}"
echo "Version=1.0" >> ${dest}/${fichier}
echo "Type=Application" >> ${dest}/${fichier}
echo "Terminal=false" >> ${dest}/${fichier}
echo "Icon=${chemin}/${nom_appli}.png" >> ${dest}/${fichier}
echo "Icon[fr_FR]=${chemin}/${nom_appli}.png" >> ${dest}/${fichier}
echo "Name[fr_FR]=${nom_appli}" >> ${dest}/${fichier}
echo "Exec=${chemin}/${nom_exec}" >> ${dest}/${fichier}
echo "Name=${nom_appli}" >> ${dest}/${fichier}

fichier="pycharm.desktop"
nom_appli="pycharm"
chemin="/opt/pycharm/bin"
nom_exec="pycharm.sh"
dest="/home/$USER/Bureau"
echo "#!/usr/bin/env xdg-open" > "${dest}/${fichier}"
echo "[Desktop Entry]" >> "${dest}/${fichier}"
echo "Version=1.0" >> ${dest}/${fichier}
echo "Type=Application" >> ${dest}/${fichier}
echo "Terminal=false" >> ${dest}/${fichier}
echo "Icon=${chemin}/${nom_appli}.png" >> ${dest}/${fichier}
echo "Exec=${chemin}/${nom_exec}" >> ${dest}/${fichier}
echo "Name=${nom_appli}" >> ${dest}/${fichier}


fichier="looping.desktop"
nom_appli="looping"
icone="/opt/looping-mcd/looping.png"
dest="/home/$USER/Bureau"
echo "#!/usr/bin/env xdg-open" > "${dest}/${fichier}"
echo "[Desktop Entry]" >> "${dest}/${fichier}"
echo "Version=1.0" >> ${dest}/${fichier}
echo "Type=Application" >> ${dest}/${fichier}
echo "Terminal=false" >> ${dest}/${fichier}
echo "Icon=${icone}" >> ${dest}/${fichier}
echo "Exec=wine /opt/looping-mcd/Looping.exe" >> ${dest}/${fichier}
