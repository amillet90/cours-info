R19 : Donnez les salaire et le nom des employés gagnant plus que tous les ingénieurs

SELECT nom, salaire FROM EMPLOYE
          WHERE salaire > ALL (SELECT salaire 
                             FROM EMPLOYE
                             WHERE fonction='INGENIEUR');
SELECT nom, salaire FROM EMPLOYE
          WHERE salaire >  (SELECT MAX(salaire) 
                             FROM EMPLOYE
                             WHERE fonction='INGENIEUR');

R20 : Donnez les noms des employés travaillant dans un département avec au moins un ingénieur

SELECT DISTINCT E2.nom, D2.nom, E2.fonction
FROM EMPLOYE E1
INNER JOIN EMPLOYE E2 ON E1.departement_id = E2.departement_id
INNER JOIN DEPARTEMENT AS D2 ON D2.idDept=E2.departement_id
WHERE E1.fonction = 'ingenieur'
AND E2.fonction <> 'ingenieur';

SELECT DISTINCT E2.nom
FROM EMPLOYE E1, EMPLOYE E2
WHERE E1.departement_id = E2.departement_id
AND E1.fonction = 'ingenieur';

R21 : Donnez les noms des employés du département “vente” embauchés le même jour qu’un employé du département “direction”

SQL :

SELECT DISTINCT E1.nom 
  FROM EMPLOYE E1, DEPARTEMENT D1, EMPLOYE E2, DEPARTEMENT D2
WHERE E1.departement_id=D1.idDept 
       AND E2.departement_id=D2.idDept 
AND D1.nom='vente'
AND D2.nom='direction'
AND E1.date_embauche=E2.date_embauche;

-- ou

SELECT EMPLOYE.nom 
FROM EMPLOYE
INNER JOIN DEPARTEMENT ON EMPLOYE.departement_id=DEPARTEMENT.idDept
AND DEPARTEMENT.nom='vente'
AND date_embauche IN (SELECT date_embauche 
                        FROM EMPLOYE
                        INNER JOIN DEPARTEMENT 
                        ON EMPLOYE.departement_id=DEPARTEMENT.idDept
                        AND DEPARTEMENT.nom='direction');

R22 : Donnez les noms des employés embauchés avant tous les employés du département “vente”.

SELECT nom
FROM EMPLOYE
WHERE date_embauche < ALL (SELECT date_embauche 
                        FROM EMPLOYE
                        INNER JOIN DEPARTEMENT 
                        ON EMPLOYE.departement_id=DEPARTEMENT.idDept
                        AND DEPARTEMENT.nom='vente');

R23 : Donnez le nom et la date d’embauche des employés embauchés avant leur reponsable; donnez également le nom et la date d’embauche dudit reponsable

SELECT E1.nom, E1.date_embauche, E2.nom, E2.date_embauche
FROM EMPLOYE E1, EMPLOYE E2
WHERE E1.idResponsable=E2.idEmploye
AND E1.date_embauche<E2.date_embauche;

R24 : Donner les noms des employés ayant le salaire maximum de chaque département.

SELECT nom,salaire
FROM EMPLOYE
WHERE (departement_id, salaire) IN (SELECT departement_id, MAX(salaire)
                        FROM EMPLOYE
                        GROUP BY departement_id);

ou

SELECT E1.nom,E1.salaire
FROM EMPLOYE AS E1
WHERE E1.salaire = (SELECT MAX(E2.salaire)
                 FROM EMPLOYE AS E2
                 WHERE E2.departement_id = E1.departement_id);

R25 : Le salaire moyen le plus bas (par profession)?

R26 : Donner les emplois ayant le salaire moyen le plus bas; donnez aussi leur salaire moyen.

SELECT AVG(salaire),fonction, min(salaire)
FROM EMPLOYE
GROUP BY fonction
HAVING AVG(salaire) = (SELECT MIN(AVG(SAL)) 
                      FROM EMP
                      GROUP BY PROF);
;

SELECT PROF, AVG(SAL)

HAVING AVG(SAL) = (SELECT MIN(AVG(SAL))

                    GROUP BY PROF);
